## Task 3: Translate ProjectPdfExport Component

**Files:**
- Modify: `src/components/wizard/ProjectPdfExport.tsx`

**Step 1: Add useTranslations import and usage**

```tsx
'use client';

import { useState } from 'react';
import { useTranslations } from 'next-intl';
import { MultiAxisPrintView } from './MultiAxisPrintView';
import type { Project } from '@/types';

interface ProjectPdfExportProps {
  project: Project;
}

export function ProjectPdfExport({ project }: ProjectPdfExportProps) {
  const [showPrintView, setShowPrintView] = useState(false);
  const t = useTranslations('pdf');

  const completedCount = project.axes.filter((a) => a.status === 'COMPLETED').length;

  if (showPrintView) {
    return (
      <MultiAxisPrintView
        project={project}
        onClose={() => setShowPrintView(false)}
      />
    );
  }

  return (
    <button
      onClick={() => setShowPrintView(true)}
      disabled={completedCount === 0}
      className="
        w-full px-4 py-2 bg-green-600 text-white rounded-md
        hover:bg-green-700 transition-colors
        disabled:bg-gray-400 disabled:cursor-not-allowed
        text-sm font-medium flex items-center justify-center gap-2
      "
    >
      <span>📄</span>
      <span>{t('exportProjectPdf')}</span>
    </button>
  );
}
```

// Helper to get default axis name based on locale
function getDefaultAxisName(index: number): string {
  // Check if we're in browser environment
  if (typeof window === 'undefined') {
    return `轴-${index}`;
  }

  // Get locale from localStorage or document
  const locale = localStorage.getItem('servo-selector-locale') || 'zh';

  if (locale === 'en') {
    return `Axis-${index}`;
  }
  return `轴-${index}`;
}

// Helper to get default project name based on locale
function getDefaultProjectName(): string {
  if (typeof window === 'undefined') {
    return '未命名项目';
  }

  const locale = localStorage.getItem('servo-selector-locale') || 'zh';

  if (locale === 'en') {
    return 'Unnamed Project';
  }
  return '未命名项目';
}
```

  "shortEn": "0.64 N·m, 3000 rpm, Type A Encoder, No Brake, Smooth Shaft"

```

**Step 2: Create script to batch update**

Create temporary script `scripts/add-en-descriptions.ts`:

```typescript
import motorsData from '../src/data/motors.json';
import fs from 'fs';

const encoderMap: Record<string, string> = {
  'A型编码器': 'Type A Encoder',
  'B型编码器': 'Type B Encoder',
  '电池盒式多圈': 'Battery Multi-turn',
  '机械式多圈': 'Mechanical Multi-turn'
};

const brakeMap: Record<string, string> = {
  '无抱闸': 'No Brake',
  '有抱闸': 'With Brake'
};

const shaftMap: Record<string, string> = {
  '光轴': 'Smooth Shaft',
  '带键轴': 'Keyed Shaft'
};

function translateDescription(zhDesc: string): string {
  let enDesc = zhDesc;

  // Replace encoder types
  Object.entries(encoderMap).forEach(([zh, en]) => {
    enDesc = enDesc.replace(zh, en);
  });

  // Replace brake info
  Object.entries(brakeMap).forEach(([zh, en]) => {
    enDesc = enDesc.replace(zh, en);
  });

  // Replace shaft type
  Object.entries(shaftMap).forEach(([zh, en]) => {
    enDesc = enDesc.replace(zh, en);
  });

  return enDesc;
}

const updatedMotors = motorsData.motors.map(motor => ({
  ...motor,
  description: {
    ...motor.description,
    shortEn: translateDescription(motor.description.short)
  }
}));

const output = {
  ...motorsData,
  motors: updatedMotors
};

fs.writeFileSync(
  'src/data/motors.json',
  JSON.stringify(output, null, 2)
);

console.log(`Updated ${updatedMotors.length} motors with English descriptions`);
```
